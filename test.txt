.data
	input_str:	.asciiz "sum 1 to :"
	output_str:	.asciiz "="
	space:		.space 20

.text

main:
		la			$a0 , input_str			#print input string
		li			$v0 , 4
		syscall

		li			$v0 , 5					#read input
		syscall

		move		$t1 , $v0				#t1 = limit (from input)

		li			$t2 , 1
Loop:
		beq			$t2 , $t1 , End_Loop

		add			$t3 , $t3 , $t2
		addi		$t2 , $t2 , 1
		jump		Loop
End_Loop:
		la		 	$a0 , output_str			#print output string
		li		 	$v0 , 4
		syscall

		li			$v0 , 1						#print output
		move		$a0 , $t3
		syscall
 end:
